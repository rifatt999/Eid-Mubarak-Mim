<script>
/* ✨ Create stars */
for(let i=0;i<100;i++){
  let star=document.createElement("div");
  star.className="star";
  star.style.left=Math.random()*100+"vw";
  star.style.top=Math.random()*100+"vh";
  star.style.animationDuration=(5+Math.random()*5)+"s";
  document.getElementById("stars").appendChild(star);
}

/* typing intro */
let text="Tasha... close your eyes 😌";
let i=0;

function typeWriter(){
  if(i<text.length){
    document.getElementById("intro").innerHTML+=text.charAt(i);
    i++;
    setTimeout(typeWriter,70);
  } else {
    setTimeout(()=>{
      document.getElementById("intro").style.display="none";
      document.getElementById("main").style.display="block";
    },1500);
  }
}
typeWriter();

/* button click */
function showLove(){
  document.getElementById("bangla").style.display="block";

  setTimeout(()=>{
    document.getElementById("loveMsg").style.display="block";
  },2000);

  setInterval(()=>{
    let heart=document.createElement("div");
    heart.className="heart";
    heart.innerHTML="❤️";
    heart.style.left=Math.random()*100+"vw";
    document.body.appendChild(heart);
    setTimeout(()=>heart.remove(),5000);
  },300);

  setTimeout(()=>{
    let answer = confirm("Will you stay with me forever? ❤️");
    if(answer){
      alert("I knew it… I love you forever Tasha 💖");
    } else {
      alert("Still… I will always love you 💔");
    }
  },5000);
}
